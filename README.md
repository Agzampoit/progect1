    answer <- matrix(c(sumSetosa, sumVersicolor, sumVirginica),

    if(answer[1,1]==0&&answer[1,2]==0&&answer[1,3]==0){
      class='white'
    }
    else{
      class = colors[which.max(answer)]
    }
    return(class)
}
